//create function to find euclidian distance between 2 arrays
CREATE OR REPLACE FUNCTION distance(l real[], r real[])
  RETURNS real AS
$BODY$
DECLARE
  s real;
BEGIN
  s := 0;
  FOR i IN 1..array_length(l, 1) LOOP
    s := s + ((l[i] - r[i]) * (l[i] - r[i]));
  END LOOP;
  RETURN |/ s;
END;
$BODY$
  LANGUAGE plpgsql VOLATILE
  COST 100;
ALTER FUNCTION distance(real[], real[])
  OWNER TO topo;
